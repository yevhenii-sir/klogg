    $ENV{KLOGG_VERSION}
    CACHE STRING "build version"
)

# Generate version from current date if not provided via environment variable
if(NOT BUILD_VERSION OR BUILD_VERSION STREQUAL "")
  string(TIMESTAMP KLOGG_AUTO_VERSION "%y.%m.%d" UTC)
  set(BUILD_VERSION "${KLOGG_AUTO_VERSION}" CACHE STRING "build version" FORCE)
  message(STATUS "Auto-generated version from build date: ${BUILD_VERSION}")
else()
  message(STATUS "Using version from environment: ${BUILD_VERSION}")
endif()

# Parse BUILD_VERSION to extract major.minor.patch for project() call
# BUILD_VERSION format is typically YY.MM.DD or YY.MM.DD.BUILD_NUMBER
string(REGEX MATCHALL "[0-9]+" _versionComponents "${BUILD_VERSION}")
list(LENGTH _versionComponents _len)

# Extract version components with defaults
if(${_len} GREATER 0)
  list(GET _versionComponents 0 _major)
else()
  set(_major "0")
endif()
if(${_len} GREATER 1)
  list(GET _versionComponents 1 _minor)
else()
  set(_minor "0")
endif()
if(${_len} GREATER 2)
  list(GET _versionComponents 2 _patch)
else()
  set(_patch "0")
endif()

# Use parsed version for project() call (will be overridden by prepare_version.cmake if BUILD_VERSION is set)
project(
  klogg
  VERSION ${_major}.${_minor}.${_patch}
  DESCRIPTION "klogg log viewer"
  LANGUAGES C CXX ASM
)

    set(CMAKE_OSX_DEPLOYMENT_TARGET ${KLOGG_OSX_DEPLOYMENT_TARGET})
    message("OSX target ${CMAKE_OSX_DEPLOYMENT_TARGET}")
  endif()

  if(NOT DEFINED CMAKE_OSX_SYSROOT OR CMAKE_OSX_SYSROOT STREQUAL "")
    execute_process(
      COMMAND xcrun --sdk macosx --show-sdk-path
      OUTPUT_VARIABLE KLOGG_OSX_SDK_PATH
      OUTPUT_STRIP_TRAILING_WHITESPACE
      RESULT_VARIABLE KLOGG_OSX_SDK_PATH_RESULT
    )
    if(KLOGG_OSX_SDK_PATH_RESULT EQUAL 0 AND NOT KLOGG_OSX_SDK_PATH STREQUAL "")
      set(CMAKE_OSX_SYSROOT "${KLOGG_OSX_SDK_PATH}" CACHE PATH "macOS SDK path" FORCE)
      message("OSX sysroot ${CMAKE_OSX_SYSROOT}")
    endif()
  endif()

  execute_process(
    COMMAND xcrun --sdk macosx --show-sdk-version
    OUTPUT_VARIABLE KLOGG_OSX_SDK_VERSION
    OUTPUT_STRIP_TRAILING_WHITESPACE
    RESULT_VARIABLE KLOGG_OSX_SDK_VERSION_RESULT
  )
  if(KLOGG_OSX_SDK_VERSION_RESULT EQUAL 0 AND NOT KLOGG_OSX_SDK_VERSION STREQUAL "")
    set(_klogg_sdk_stamp "${CMAKE_BINARY_DIR}/.klogg_sdk_version")
    if(EXISTS "${_klogg_sdk_stamp}")
      file(READ "${_klogg_sdk_stamp}" _klogg_sdk_prev)
      string(STRIP "${_klogg_sdk_prev}" _klogg_sdk_prev)
      if(NOT _klogg_sdk_prev STREQUAL KLOGG_OSX_SDK_VERSION)
        message(FATAL_ERROR
                "macOS SDK version changed from ${_klogg_sdk_prev} to ${KLOGG_OSX_SDK_VERSION}. "
                "Please delete your build directory to avoid mixed SDK objects.")
      endif()
    endif()
    file(WRITE "${_klogg_sdk_stamp}" "${KLOGG_OSX_SDK_VERSION}\n")
  endif()
endif()
